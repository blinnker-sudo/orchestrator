# Generar UUID del cluster
./bin/windows/kafka-storage.bat random-uuid


# Formatear el storage con ese UUID
./bin/windows/kafka-storage.bat format -t <UUID-AQUI> -c config/kraft/server.properties


./bin/windows/kafka-server-start.bat config/kraft/server.properties
